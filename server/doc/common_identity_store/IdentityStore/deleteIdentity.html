void deleteIdentity(Identity identity) {
  if (_lookup(identity, 'Deletion') == null) return;

  _idStore.remove(identity.username);
  saveToDisk();

  _logger.log(
    'Deleted identity for username: ${identity.username}',
    'IdentityStore/deleteIdentity',
    LogLevel.info,
  );
}
